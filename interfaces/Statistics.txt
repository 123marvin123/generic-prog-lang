namespace Statistics {
    fun mean(seq: Seq::Seq) -> Number::Number {
        description { "arithmetic mean of elements of p(1)" }
        requires { Object::isModelOf(Seq::item(seq), Number::Number) }
        generic() {
            Number::div(
                Number::sum(seq),
                Seq::length(seq)
            )
        }
    }

    fun variance(set: Set::Set) -> Object::Object {
      description {
        "variance of elements of multi-set p(1)"
      }
    }
}
